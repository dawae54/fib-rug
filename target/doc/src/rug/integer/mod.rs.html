17/*!
18Arbitrary-precision integers.
19
20This module provides support for arbitrary-precision integers of type
21[`Integer`]. Instances of [`Integer`] always have a heap allocation for the bit
22data; if you want a temporary small integer without heap allocation, you can use
23the [`MiniInteger`] type.
24
25# Examples
26
27```rust
28use rug::integer::MiniInteger;
29use rug::Assign;
30use rug::Integer;
31let mut int = Integer::from(10);
32assert_eq!(int, 10);
33let small = MiniInteger::from(-15);
34// `small` can be borrowed like an `Integer` in the following line:
35int.assign(small.borrow().abs_ref());
36assert_eq!(int, 15);
37```
38
39[`Integer`]: crate::Integer
40*/

94/**
95The ordering of digits inside a [slice], and bytes inside a digit.
96
97# Examples
98
99```rust
100use rug::integer::Order;
101use rug::Integer;
102
103let i = Integer::from(0x0102_0304);
104let mut buf: [u16; 4] = [0; 4];
105
106// most significant 16-bit digit first, little endian digits
107i.write_digits(&mut buf, Order::MsfLe);
108assert_eq!(buf, [0, 0, 0x0102u16.to_le(), 0x0304u16.to_le()]);
109// least significant 16-bit digit first, big endian digits
110i.write_digits(&mut buf, Order::LsfBe);
111assert_eq!(buf, [0x0304u16.to_be(), 0x0102u16.to_be(), 0, 0]);
112```
113*/
114#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
115pub enum Order {
116    /// Least significant digit first, with each digit in the target’s
117    /// endianness.
118    Lsf,
119    /// Least significant digit first, with little endian digits.
120    LsfLe,
121    /// Least significant digit first, with big endian digits.
122    LsfBe,
123    /// Most significant digit first, with each digit in the target’s
124    /// endianness.
125    Msf,
126    /// Most significant digit first, with little endian digits.
127    MsfLe,
128    /// Most significant digit first, with big endian digits.
129    MsfBe,
130}
131
132impl Order {
133    #[inline]
134    fn order(self) -> c_int {
135        match self {
136            Order::Lsf | Order::LsfLe | Order::LsfBe => -1,
137            Order::Msf | Order::MsfLe | Order::MsfBe => 1,
138        }
139    }
140    #[inline]
141    fn endian(self) -> c_int {
142        match self {
143            Order::Lsf | Order::Msf => 0,
144            Order::LsfLe | Order::MsfLe => -1,
145            Order::LsfBe | Order::MsfBe => 1,
146        }
147    }
148}
