39pub trait OptFloat: Copy {
40    const IS_SOME: bool;
41    fn mpfr(self) -> *const mpfr_t;
42    fn mpfr_or(self, default: *mut mpfr_t) -> *const mpfr_t;
43    #[cfg(feature = "rational")]
44    fn unwrap_or<'a>(self, default: &'a mut Float) -> &'a Float
45    where
46        Self: 'a;
47}
48
49impl OptFloat for () {
50    const IS_SOME: bool = false;
51    #[inline(always)]
52    fn mpfr(self) -> *const mpfr_t {
53        unreachable!("unwrapping ()");
54    }
55    #[inline(always)]
56    fn mpfr_or(self, default: *mut mpfr_t) -> *const mpfr_t {
57        default.cast_const()
58    }
59    #[cfg(feature = "rational")]
60    #[inline(always)]
61    fn unwrap_or<'a>(self, default: &'a mut Float) -> &'a Float
62    where
63        Self: 'a,
64    {
65        &*default
66    }
67}
68
69impl OptFloat for &Float {
70    const IS_SOME: bool = true;
71    #[inline(always)]
72    fn mpfr(self) -> *const mpfr_t {
73        self.as_raw()
74    }
75    #[inline(always)]
76    fn mpfr_or(self, _default: *mut mpfr_t) -> *const mpfr_t {
77        self.as_raw()
78    }
79    #[cfg(feature = "rational")]
80    #[inline(always)]
81    fn unwrap_or<'b>(self, _default: &'b mut Float) -> &'b Float
82    where
83        Self: 'b,
84    {
85        self
86    }
87}

89#[inline]
90pub fn raw_round(round: Round) -> rnd_t {
91    match round {
92        Round::Nearest => rnd_t::RNDN,
93        Round::Zero => rnd_t::RNDZ,
94        Round::Up => rnd_t::RNDU,
95        Round::Down => rnd_t::RNDD,
96        Round::AwayZero => rnd_t::RNDA,
97    }
98}

100#[inline]
101pub fn ordering1(ord: c_int) -> Ordering {
102    ord.cmp(&0)
103}
104
105#[inline]
106fn ordering2(ord: c_int) -> (Ordering, Ordering) {
107    // ord == first + 4 * second
108    let first = match ord & 3 {
109        2 => -1,
110        0 => 0,
111        1 => 1,
112        _ => unreachable!(),
113    };
114    let second = match ord >> 2 {
115        2 => -1,
116        0 => 0,
117        1 => 1,
118        _ => unreachable!(),
119    };
120    (ordering1(first), ordering1(second))
121}

123macro_rules! unsafe_wrap {
124    (fn $fn:ident($($op:ident: $O:ident),* $(; $param:ident: $T:ty)*) -> $deleg:path) => {
125        #[inline]
126        pub fn $fn<$($O: OptFloat),*>(
127            rop: &mut Float,
128            $($op: $O,)*
129            $($param: $T,)*
130            rnd: Round,
131        ) -> Ordering {
132            let rop = rop.as_raw_mut();
133            $(let $op = $op.mpfr_or(rop);)*
134            ordering1(unsafe { $deleg(rop, $($op,)* $($param.into(),)* raw_round(rnd)) })
135        }
136    };
137}
138
139macro_rules! unsafe_wrap0 {
140    (fn $fn:ident($($param:ident: $T:ty),*) -> $deleg:path) => {
141        #[inline]
142        pub fn $fn(rop: &mut Float, $($param: $T,)* rnd: Round) -> Ordering {
143            ordering1(unsafe { $deleg(rop.as_raw_mut(), $($param.into(),)* raw_round(rnd)) })
144        }
145    };
146}

148#[inline]
149pub fn si_pow_ui(rop: &mut Float, base: i32, exponent: u32, rnd: Round) -> Ordering {
150    let (base_neg, base_abs) = base.neg_abs();
151    if !base_neg || (exponent & 1) == 0 {
152        ordering1(unsafe {
153            mpfr::ui_pow_ui(
154                rop.as_raw_mut(),
155                base_abs.into(),
156                exponent.into(),
157                raw_round(rnd),
158            )
159        })
160    } else {
161        let reverse_rnd = raw_round(rnd.reverse());
162        let reverse_ord = ordering1(unsafe {
163            mpfr::ui_pow_ui(
164                rop.as_raw_mut(),
165                base_abs.into(),
166                exponent.into(),
167                reverse_rnd,
168            )
169        });
170        neg(rop, (), Round::Nearest);
171        reverse_ord.reverse()
172    }
173}

175#[inline]
176pub fn write_new_nan(x: &mut MaybeUninit<Float>, prec: prec_t) {
177    // Safety: we can cast pointers to/from Float/mpfr_t as they are repr(transparent).
178    unsafe {
179        let inner_ptr = misc::cast_ptr_mut(x.as_mut_ptr());
180        mpfr::init2(inner_ptr, prec);
181    }
182}

184// do not use mpfr::neg for op is () to avoid function call
185#[inline]
186pub fn neg<O: OptFloat>(rop: &mut Float, op: O, rnd: Round) -> Ordering {
187    if O::IS_SOME {
188        ordering1(unsafe { mpfr::neg(rop.as_raw_mut(), op.mpfr(), raw_round(rnd)) })
189    } else {
190        // Safety: negating the sign of mpfr_t is always sound.
191        unsafe {
192            rop.inner_mut().sign.neg_assign();
193        }
194        if rop.is_nan() {
195            set_nanflag();
196        }
197        Ordering::Equal
198    }
199}

201#[inline]
202pub fn signum<O: OptFloat>(rop: &mut Float, op: O, _rnd: Round) -> Ordering {
203    let rop = rop.as_raw_mut();
204    let op = op.mpfr_or(rop);
205    unsafe {
206        if mpfr::nan_p(op) != 0 {
207            mpfr::set(rop, op, rnd_t::RNDZ);
208        } else if mpfr::signbit(op) != 0 {
209            mpfr::set_si(rop, -1, rnd_t::RNDZ);
210        } else {
211            mpfr::set_si(rop, 1, rnd_t::RNDZ);
212        }
213    }
214    // mpfr_t has a minimum precision of 1, so -1 and 1 are exactly representable
215    Ordering::Equal
216}

278#[inline]
279pub fn lgamma<O: OptFloat>(rop: &mut Float, op: O, rnd: Round) -> (Ordering, Ordering) {
280    let mut sign = MaybeUninit::uninit();
281    let rop = rop.as_raw_mut();
282    let op = op.mpfr_or(rop);
283    let ord = ordering1(unsafe { mpfr::lgamma(rop, sign.as_mut_ptr(), op, raw_round(rnd)) });
284    let sign_ord = if (unsafe { sign.assume_init() }) < 0 {
285        Ordering::Less
286    } else {
287        Ordering::Greater
288    };
289    (sign_ord, ord)
290}

292pub fn sum<'a, I>(rop: &mut Float, values: I, rnd: Round) -> Ordering
293where
294    I: Iterator<Item = &'a Float>,
295{
296    let pointers = values.map(Float::as_raw).collect::<VecLike<_>>();
297    unsafe { sum_raw(rop.as_raw_mut(), pointers.as_slice(), rnd) }
298}

300// add original value of rop to sum
301pub fn sum_including_old<'a, I>(rop: &mut Float, values: I, rnd: Round) -> Ordering
302where
303    I: Iterator<Item = &'a Float>,
304{
305    let rop = rop.as_raw_mut();
306    let capacity = values.size_hint().0.checked_add(1).expect("overflow");
307    let mut pointers = VecLike::new();
308    pointers.reserve(capacity);
309    pointers.push(rop.cast_const());
310    pointers.extend(values.map(Float::as_raw));
311    unsafe { sum_raw(rop, pointers.as_slice(), rnd) }
312}

314pub unsafe fn sum_raw(rop: *mut mpfr_t, pointers: &[*const mpfr_t], rnd: Round) -> Ordering {
315    let n = pointers.len().unwrapped_cast();
316    let tab = misc::cast_ptr(pointers.as_ptr());
317    let rnd = raw_round(rnd);
318    ordering1(unsafe { mpfr::sum(rop, tab, n, rnd) })
319}

321pub fn dot<'a, I>(rop: &mut Float, values: I, rnd: Round) -> Ordering
322where
323    I: Iterator<Item = (&'a Float, &'a Float)>,
324{
325    let (pointers_a, pointers_b): (VecLike<_>, VecLike<_>) =
326        values.map(|(a, b)| (a.as_raw(), b.as_raw())).unzip();
327    unsafe {
328        dot_raw(
329            rop.as_raw_mut(),
330            pointers_a.as_slice(),
331            pointers_b.as_slice(),
332            rnd,
333        )
334    }
335}

337// add original value of rop to dot
338pub fn dot_including_old<'a, I>(rop: &mut Float, values: I, rnd: Round) -> Ordering
339where
340    I: Iterator<Item = (&'a Float, &'a Float)>,
341{
342    const LIMB_ONE: limb_t = 1;
343    const LIMB_MSB: limb_t = LIMB_ONE << (gmp::LIMB_BITS - 1);
344    const ONE: mpfr_t = mpfr_t {
345        prec: 1,
346        sign: 1,
347        exp: 1,
348        d: unsafe { NonNull::new_unchecked((&LIMB_MSB as *const limb_t).cast_mut()) },
349    };
350
351    let rop = rop.as_raw_mut();
352    let capacity = values.size_hint().0.checked_add(1).expect("overflow");
353    let mut pointers_a = VecLike::new();
354    let mut pointers_b = VecLike::new();
355    pointers_a.reserve(capacity);
356    pointers_b.reserve(capacity);
357    pointers_a.push(rop.cast_const());
358    pointers_b.push(&ONE as *const mpfr_t);
359    for a_b in values {
360        pointers_a.push(a_b.0.as_raw());
361        pointers_b.push(a_b.1.as_raw());
362    }
363    unsafe { dot_raw(rop, pointers_a.as_slice(), pointers_b.as_slice(), rnd) }
364}
365
366// pointers_a and pointers_b must have same length
367unsafe fn dot_raw(
368    rop: *mut mpfr_t,
369    pointers_a: &[*const mpfr_t],
370    pointers_b: &[*const mpfr_t],
371    rnd: Round,
372) -> Ordering {
373    debug_assert_eq!(pointers_a.len(), pointers_b.len());
374    let n = pointers_a.len().unwrapped_cast();
375    let a = misc::cast_ptr(pointers_a.as_ptr());
376    let b = misc::cast_ptr(pointers_b.as_ptr());
377    let rnd = raw_round(rnd);
378    ordering1(unsafe { mpfr::dot(rop, a, b, n, rnd) })
379}

514#[inline]
515pub fn remainder_quo31<O: OptFloat, P: OptFloat>(
516    rop: &mut Float,
517    op1: O,
518    op2: P,
519    rnd: Round,
520) -> (Ordering, i32) {
521    let rop = rop.as_raw_mut();
522    let op1 = op1.mpfr_or(rop);
523    let op2 = op2.mpfr_or(rop);
524    let ord;
525    let quo31;
526    unsafe {
527        let mut quo_long = MaybeUninit::<c_long>::uninit();
528        ord = mpfr::remquo(rop, quo_long.as_mut_ptr(), op1, op2, raw_round(rnd));
529        quo31 = if mpfr::nan_p(rop) == 0 {
530            let quo_long = quo_long.assume_init();
531            let lower31 = quo_long.wrapping_as::<i32>() & i32::MAX;
532            // Do not add sign if lower31 is 0, otherwise we end up with
533            // i32::MIN which has one extra significant bit.
534            let sign = (quo_long.is_negative() && lower31 != 0).wrapping_as::<i32>() << 31;
535            sign | lower31
536        } else {
537            0
538        };
539    }
540    (ordering1(ord), quo31)
541}

543#[inline]
544pub fn frexp<O: OptFloat>(rop: &mut Float, op: O, rnd: Round) -> (Ordering, i32) {
545    let rop = rop.as_raw_mut();
546    let op = op.mpfr_or(rop);
547    let ord;
548    let exp32;
549    unsafe {
550        let mut exp = MaybeUninit::<exp_t>::uninit();
551        ord = mpfr::frexp(exp.as_mut_ptr(), rop, op, raw_round(rnd));
552        exp32 = if mpfr::number_p(rop) != 0 {
553            let exp = exp.assume_init();
554            exp.unwrapped_cast()
555        } else {
556            0
557        };
558    }
559    (ordering1(ord), exp32)
560}

625    // Safety: We do not change inner.d, and we set inner.exp to
626    // indicate a singular number so even if the data at inner.d is

631            Special::Zero => {

634            }
635            Special::NegZero => {

642            }
643            Special::NegInfinity => {

645                inner.exp = EXP_INF;
646            }
647            Special::Nan => {
648                inner.sign = 1;
649                inner.exp = EXP_NAN;
650            }
651        }
652    }

718#[cfg(feature = "nightly-float")]
719pub fn get_f16(op: &Float, rnd: Round) -> f16 {
720    const ZERO: MiniFloat = MiniFloat::const_from_f16(0.0);
721    let mut small = ZERO;
722    // SAFETY: we are not going to change precision.
723    let small_mut = unsafe { small.as_nonreallocating_float() };
724    let ordering = set(small_mut, op, rnd);
725    let negative = small_mut.is_sign_negative();
726    if small_mut.is_nan() {
727        let n = f16::NAN;
728        return if negative == n.is_sign_negative() {
729            n
730        } else {
731            -n
732        };
733    }
734    if small_mut.is_infinite() {
735        return if negative {
736            f16::NEG_INFINITY
737        } else {
738            f16::INFINITY
739        };
740    }
741    if small_mut.is_zero() {
742        return if negative { -0.0 } else { 0.0 };
743    }
744    let exp = small_mut.get_exp().expect("expected exponent");
745    if exp > f16::MAX_EXP {
746        return if negative {
747            f16::NEG_INFINITY
748        } else {
749            f16::INFINITY
750        };
751    }
752    let sign = if negative { 1u16 << 15 } else { 0 };
753    if exp < f16::MIN_EXP {
754        // Subnormal.
755        small_mut.subnormalize_round(f16::MIN_EXP, ordering, rnd);
756    }
757    // Subnormalization may change exp because of rounding, so update it.
758    if small_mut.is_zero() {
759        return if negative { -0.0 } else { 0.0 };
760    }
761    let exp = small_mut.get_exp().expect("expected exponent");
762    if exp < f16::MIN_EXP {
763        // Still subnormal.
764
765        // The number of significant bits is MANTISSA_DIGITS - (MIN_EXP - exp).
766        // For example, if we are just subnormal; then exp == MIN_EXP - 1, and
767        // we should have MANTISSA_DIGITS - 1 significant bits.
768        let diff = f16::MIN_EXP.abs_diff(exp);
769        if diff >= f16::MANTISSA_DIGITS {
770            return if negative { -0.0 } else { 0.0 };
771        }
772        let nsig = f16::MANTISSA_DIGITS - diff;
773        let mant = (small_mut.inner_data()[0] >> (limb_t::BITS - nsig)) as u16;
774
775        return f16::from_bits(sign | mant);
776    }
777    // Normal. MANTISSA_DIGITS significant digits.
778    let mant = (small_mut.inner_data()[0] >> (limb_t::BITS - f16::MANTISSA_DIGITS)) as u16;
779    // This includes the implicit one, which acts as 1 in the exponent.
780    // So we need to add (bias - 1) instead of bias to the exponent.
781    // For example, if exp == MIN_EXP, we need to add 0, since the biased
782    // exponent should be 1 for the smallest normal number.
783    let biased_minus_1 = (exp - f16::MIN_EXP) as u16;
784    let exp_mant = (biased_minus_1 << (f16::MANTISSA_DIGITS - 1)) + mant;
785    f16::from_bits(sign + exp_mant)
786}

797#[cfg(feature = "nightly-float")]
798pub fn get_f128(op: &Float, rnd: Round) -> f128 {
799    const ZERO: MiniFloat = MiniFloat::const_from_f128(0.0);
800    let mut small = ZERO;
801    // SAFETY: we are not going to change precision.
802    let small_mut = unsafe { small.as_nonreallocating_float() };
803    let ordering = set(small_mut, op, rnd);
804    let negative = small_mut.is_sign_negative();
805    if small_mut.is_nan() {
806        let n = f128::NAN;
807        return if negative == n.is_sign_negative() {
808            n
809        } else {
810            -n
811        };
812    }
813    if small_mut.is_infinite() {
814        return if negative {
815            f128::NEG_INFINITY
816        } else {
817            f128::INFINITY
818        };
819    }
820    if small_mut.is_zero() {
821        return if negative { -0.0 } else { 0.0 };
822    }
823    let exp = small_mut.get_exp().expect("expected exponent");
824    if exp > f128::MAX_EXP {
825        return if negative {
826            f128::NEG_INFINITY
827        } else {
828            f128::INFINITY
829        };
830    }
831    let sign = if negative { 1u128 << 127 } else { 0 };
832    if exp < f128::MIN_EXP {
833        // Subnormal.
834        small_mut.subnormalize_round(f128::MIN_EXP, ordering, rnd);
835    }
836    // Subnormalization may change exp because of rounding, so update it.
837    if small_mut.is_zero() {
838        return if negative { -0.0 } else { 0.0 };
839    }
840    let exp = small_mut.get_exp().expect("expected exponent");
841    if exp < f128::MIN_EXP {
842        // Still subnormal.
843
844        // The number of significant bits is MANTISSA_DIGITS - (MIN_EXP - exp).
845        // For example, if we are just subnormal; then exp == MIN_EXP - 1, and
846        // we should have MANTISSA_DIGITS - 1 significant bits.
847        let diff = f128::MIN_EXP.abs_diff(exp);
848        if diff >= f128::MANTISSA_DIGITS {
849            return if negative { -0.0 } else { 0.0 };
850        }
851        let nsig = f128::MANTISSA_DIGITS - diff;
852        let mant = {
853            let data = small_mut.inner_data();
854            #[cfg(gmp_limb_bits_64)]
855            {
856                ((data[1] as u128) << 64) | (data[0] as u128)
857            }
858            #[cfg(gmp_limb_bits_32)]
859            {
860                ((data[3] as u128) << 96)
861                    | ((data[2] as u128) << 64)
862                    | ((data[1] as u128) << 32)
863                    | (data[0] as u128)
864            }
865        } >> (128 - nsig);
866
867        return f128::from_bits(sign | mant);
868    }
869    // Normal. MANTISSA_DIGITS significant digits.
870    let mant = {
871        let data = small_mut.inner_data();
872        #[cfg(gmp_limb_bits_64)]
873        {
874            ((data[1] as u128) << 64) | (data[0] as u128)
875        }
876        #[cfg(gmp_limb_bits_32)]
877        {
878            ((data[3] as u128) << 96)
879                | ((data[2] as u128) << 64)
880                | ((data[1] as u128) << 32)
881                | (data[0] as u128)
882        }
883    } >> (128 - f128::MANTISSA_DIGITS);
884    // This includes the implicit one, which acts as 1 in the exponent.
885    // So we need to add (bias - 1) instead of bias to the exponent.
886    // For example, if exp == MIN_EXP, we need to add 0, since the biased
887    // exponent should be 1 for the smallest normal number.
888    let biased_minus_1 = (exp - f128::MIN_EXP) as u128;
889    let exp_mant = (biased_minus_1 << (f128::MANTISSA_DIGITS - 1)) + mant;
890    f128::from_bits(sign + exp_mant)
891}

1077pub fn submul<O: OptFloat>(
1078    rop: &mut Float,
1079    add: O,
1080    mul1: &Float,
1081    mul2: &Float,
1082    rnd: Round,
1083) -> Ordering {
1084    let reverse_ord = fms(rop, mul1, mul2, add, rnd.reverse());

1273pub fn q_div<O: OptFloat>(rop: &mut Float, op1: &Rational, op2: O, rnd: Round) -> Ordering {
1274    let denom = {
1275        let op1_den = op1.denom();
1276        let op2 = op2.unwrap_or(rop);
1277        let prec = op1_den
1278            .significant_bits()
1279            .checked_add(op2.prec())
1280            .expect("overflow");
1281        Float::with_val(prec, op1_den * op2)
1282    };
1283    z_div(rop, op1.numer(), &denom, rnd)
1284}
