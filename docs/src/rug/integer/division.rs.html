704trait AsLong: Copy {
705    type Long;
706}
707
708macro_rules! as_long {
709    ($Long:ty: $($Prim:ty)*) => { $(
710        impl AsLong for $Prim {
711            type Long = $Long;
712        }
713    )* }
714}
715
716as_long! { c_long: i8 i16 i32 i64 i128 isize }
717as_long! { c_ulong: u8 u16 u32 u64 u128 usize }
718
719macro_rules! forward {
720    (fn $fn:ident() -> $deleg_long:path, $deleg:path) => {
721        #[inline]
722        fn $fn<O: OptInteger>(rop: &mut Integer, op1: O, op2: Self) {
723            if let Some(op2) = op2.checked_as() {
724                $deleg_long(rop, op1, op2);
725            } else {
726                let mut small: MiniInteger = op2.into();
727                $deleg(rop, op1, small.borrow_excl());
728            }
729        }
730    };
731}
732macro_rules! reverse {
733    (fn $fn:ident() -> $deleg_long:path, $deleg:path) => {
734        #[inline]
735        fn $fn<O: OptInteger>(rop: &mut Integer, op1: Self, op2: O) {
736            if let Some(op1) = op1.checked_as() {
737                $deleg_long(rop, op1, op2);
738            } else {
739                let mut small: MiniInteger = op1.into();
740                $deleg(rop, small.borrow_excl(), op2);
741            }
742        }
743    };
744}

867    #[test]
868    fn check_trunc() {
869        let ndqr = [
870            (23, 10, 2, 3),
871            (23, -10, -2, 3),
872            (-23, 10, -2, -3),
873            (-23, -10, 2, -3),
874            (20, 10, 2, 0),
875            (20, -10, -2, 0),
876            (-20, 10, -2, 0),
877            (-20, -10, 2, 0),
878            (3, 10, 0, 3),
879            (3, -10, 0, 3),
880            (-3, 10, 0, -3),
881            (-3, -10, 0, -3),
882            (0, 10, 0, 0),
883            (0, -10, 0, 0),
884        ];
885        for &(n, d, q, r) in &ndqr {
886            assert_eq!(Integer::from(n) / d, q);
887            assert_eq!(Integer::from(n).div_trunc(d), q);
888            assert_eq!(Integer::from(n) % d, r);
889            assert_eq!(Integer::from(n).rem_trunc(d), r);
890            let qr = Integer::from(n).div_rem(Integer::from(d));
891            assert_eq!(qr.0, q);
892            assert_eq!(qr.1, r);
893            let (mut nq, mut dr) = (Integer::from(n), Integer::from(d));
894            let qr = <(Integer, Integer)>::from(nq.div_rem_ref(&dr));
895            assert_eq!(qr.0, q);
896            assert_eq!(qr.1, r);
897            nq.div_rem_mut(&mut dr);
898            assert_eq!(nq, q);
899            assert_eq!(dr, r);
900        }
901    }
