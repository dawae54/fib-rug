17use crate::Complex;
18use core::cmp::Ordering;
19use core::hash::{Hash, Hasher};
20
21/**
22A complex number that supports total ordering and hashing.
23
24For ordering, the real part has precedence over the imaginary part. Negative
25zero is ordered as less than positive zero. Negative NaN is ordered as less than
26negative infinity, while positive NaN is ordered as greater than positive
27infinity. Comparing two negative NaNs or two positive NaNs produces equality.
28
29# Examples
30
31```rust
32use core::cmp::Ordering;
33use rug::complex::OrdComplex;
34use rug::float::Special;
35use rug::Complex;
36
37let nan_c = Complex::with_val(53, (Special::Nan, Special::Nan));
38let nan = OrdComplex::from(nan_c);
39assert_eq!(nan.cmp(&nan), Ordering::Equal);
40
41let one_neg0_c = Complex::with_val(53, (1, Special::NegZero));
42let one_neg0 = OrdComplex::from(one_neg0_c);
43let one_pos0_c = Complex::with_val(53, (1, Special::Zero));
44let one_pos0 = OrdComplex::from(one_pos0_c);
45assert_eq!(one_neg0.cmp(&one_pos0), Ordering::Less);
46
47let zero_inf_s = (Special::Zero, Special::Infinity);
48let zero_inf_c = Complex::with_val(53, zero_inf_s);
49let zero_inf = OrdComplex::from(zero_inf_c);
50assert_eq!(one_pos0.cmp(&zero_inf), Ordering::Greater);
51```
52*/
53#[derive(Clone, Debug)]
54#[repr(transparent)]
55pub struct OrdComplex {
56    inner: Complex,
57}
58
59static_assert_same_layout!(OrdComplex, Complex);
60
61impl OrdComplex {
62    /// Extracts the underlying [`Complex`].
63    ///
64    /// The same result can be obtained using the implementation of
65    /// <code>[AsRef]\<[Complex]></code> which is provided for [`OrdComplex`].
66    ///
67    /// # Examples
68    ///
69    /// ```rust
70    /// use rug::complex::OrdComplex;
71    /// use rug::Complex;
72    /// let c = Complex::with_val(53, (1.5, 2.5));
73    /// let ord = OrdComplex::from(c);
74    /// let c_ref = ord.as_complex();
75    /// assert_eq!(*c_ref.real(), 1.5);
76    /// assert_eq!(*c_ref.imag(), 2.5);
77    /// ```
78    #[inline]
79    pub const fn as_complex(&self) -> &Complex {
80        &self.inner
81    }
82
83    /// Extracts the underlying [`Complex`].
84    ///
85    /// The same result can be obtained using the implementation of
86    /// <code>[AsMut]\<[Complex]></code> which is provided for [`OrdComplex`].
87    ///
88    /// # Examples
89    ///
90    /// ```rust
91    /// use rug::complex::OrdComplex;
92    /// use rug::Complex;
93    /// let c = Complex::with_val(53, (1.5, -2.5));
94    /// let mut ord = OrdComplex::from(c);
95    /// ord.as_complex_mut().conj_mut();
96    /// let c_ref = ord.as_complex();
97    /// assert_eq!(*c_ref.real(), 1.5);
98    /// assert_eq!(*c_ref.imag(), 2.5);
99    /// ```
100    #[inline]
101    pub fn as_complex_mut(&mut self) -> &mut Complex {
102        &mut self.inner
103    }
104}
105
106impl Hash for OrdComplex {
107    #[inline]
108    fn hash<H: Hasher>(&self, state: &mut H) {
109        self.inner.real().as_ord().hash(state);
110        self.inner.imag().as_ord().hash(state);
111    }
112}
113
114impl Eq for OrdComplex {}
115
116impl Ord for OrdComplex {
117    #[inline]
118    fn cmp(&self, other: &OrdComplex) -> Ordering {
119        self.inner.total_cmp(&other.inner)
120    }
121}
122
123impl PartialEq for OrdComplex {
124    #[inline]
125    fn eq(&self, other: &OrdComplex) -> bool {
126        self.inner.real().as_ord().eq(other.inner.real().as_ord())
127            && self.inner.imag().as_ord().eq(other.inner.imag().as_ord())
128    }
129}
130
131impl PartialOrd for OrdComplex {
132    #[inline]
133    fn partial_cmp(&self, other: &OrdComplex) -> Option<Ordering> {
134        Some(self.cmp(other))
135    }
136}
137
138impl From<Complex> for OrdComplex {
139    #[inline]
140    fn from(src: Complex) -> Self {
141        OrdComplex { inner: src }
142    }
143}
144
145impl From<OrdComplex> for Complex {
146    #[inline]
147    fn from(src: OrdComplex) -> Self {
148        src.inner
149    }
150}
151
152impl AsRef<Complex> for OrdComplex {
153    #[inline]
154    fn as_ref(&self) -> &Complex {
155        self.as_complex()
156    }
157}
158
159impl AsMut<Complex> for OrdComplex {
160    #[inline]
161    fn as_mut(&mut self) -> &mut Complex {
162        self.as_complex_mut()
163    }
164}

166#[allow(clippy::eq_op)]
167#[cfg(test)]
168mod tests {
169    use crate::complex::OrdComplex;
170    use crate::float;
171    use crate::float::{FreeCache, Special};
172    use crate::Complex;
173    use core::hash::{Hash, Hasher};
174    use core::ptr;
175    use std::collections::hash_map::DefaultHasher;
176
177    fn calculate_hash<T: Hash>(t: &T) -> u64 {
178        let mut s = DefaultHasher::new();
179        t.hash(&mut s);
180        s.finish()
181    }
182
183    #[test]
184    fn check_zero() {
185        let pp = Complex::with_val(53, (Special::Zero, Special::Zero));
186        let pn = Complex::with_val(53, (Special::Zero, Special::NegZero));
187        let np = Complex::with_val(53, (Special::NegZero, Special::Zero));
188        let nn = Complex::with_val(53, (Special::NegZero, Special::NegZero));
189        assert_eq!(pp, pn);
190        assert_eq!(pn, np);
191        assert_eq!(np, nn);
192        assert_eq!(nn, pp);
193        let ord_pp = pp.as_ord();
194        let ord_pn = pn.as_ord();
195        let ord_np = np.as_ord();
196        let ord_nn = nn.as_ord();
197        assert_eq!(ord_pp, ord_pp);
198        assert_eq!(ord_pn, ord_pn);
199        assert_eq!(ord_np, ord_np);
200        assert_eq!(ord_nn, ord_nn);
201        assert_eq!(calculate_hash(ord_pp), calculate_hash(ord_pp));
202        assert_eq!(calculate_hash(ord_pn), calculate_hash(ord_pn));
203        assert_eq!(calculate_hash(ord_np), calculate_hash(ord_np));
204        assert_eq!(calculate_hash(ord_nn), calculate_hash(ord_nn));
205        assert_ne!(ord_pp, ord_pn);
206        assert_ne!(ord_pn, ord_np);
207        assert_ne!(ord_np, ord_nn);
208        assert_ne!(ord_nn, ord_pp);
209        assert_ne!(calculate_hash(ord_pp), calculate_hash(ord_pn));
210        assert_ne!(calculate_hash(ord_pn), calculate_hash(ord_np));
211        assert_ne!(calculate_hash(ord_np), calculate_hash(ord_nn));
212        assert_ne!(calculate_hash(ord_nn), calculate_hash(ord_pp));
213
214        float::free_cache(FreeCache::All);
215    }

217    #[test]
218    fn check_refs() {
219        let f = Complex::with_val(53, (23.5, 32.5));
220        assert_eq!(
221            ptr::addr_of!(f).cast::<OrdComplex>(),
222            f.as_ord() as *const OrdComplex
223        );
224        assert_eq!(
225            ptr::addr_of!(f).cast::<OrdComplex>(),
226            AsRef::<OrdComplex>::as_ref(&f) as *const OrdComplex
227        );
228        let mut o = OrdComplex::from(f);
229        assert_eq!(
230            ptr::addr_of!(o).cast::<Complex>(),
231            o.as_complex() as *const Complex
232        );
233        assert_eq!(
234            ptr::addr_of!(o).cast::<Complex>(),
235            AsRef::<Complex>::as_ref(&o) as *const Complex
236        );
237        assert_eq!(
238            ptr::addr_of_mut!(o).cast::<Complex>(),
239            o.as_complex_mut() as *mut Complex
240        );
241        assert_eq!(
242            ptr::addr_of_mut!(o).cast::<Complex>(),
243            AsMut::<Complex>::as_mut(&mut o) as *mut Complex
244        );
245    }
