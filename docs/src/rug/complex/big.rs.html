45/**
46A multi-precision complex number with arbitrarily large precision and correct
47rounding.
48
49The precision has to be set during construction. The rounding method of the
50required operations can be specified, and the direction of the rounding is
51returned.
52
53# Examples
54
55```rust
56use rug::{Assign, Complex, Float};
57let c = Complex::with_val(53, (40, 30));
58assert_eq!(format!("{:.3}", c), "(40.0 30.0)");
59let mut f = Float::with_val(53, c.abs_ref());
60assert_eq!(f, 50);
61f.assign(c.arg_ref());
62assert_eq!(f, 0.75_f64.atan());
63```
64
65Operations on two borrowed `Complex` numbers result in an
66[incomplete-computation value][icv] that has to be assigned to a new `Complex`
67number.
68
69```rust
70use rug::Complex;
71let a = Complex::with_val(53, (10.5, -11));
72let b = Complex::with_val(53, (-1.25, -1.5));
73let a_b_ref = &a + &b;
74let a_b = Complex::with_val(53, a_b_ref);
75assert_eq!(a_b, (9.25, -12.5));
76```
77
78As a special case, when an [incomplete-computation value][icv] is obtained from
79multiplying two `Complex` number references, it can be added to or subtracted
80from another `Complex` number (or reference). This will result in a fused
81multiply-accumulate operation, with only one rounding operation taking place.
82
83```rust
84use rug::Complex;
85let mut acc = Complex::with_val(53, (1000, 1000));
86let m1 = Complex::with_val(53, (10, 0));
87let m2 = Complex::with_val(53, (1, -1));
88// (1000 + 1000i) - (10 + 0i) × (1 - i) = (990 + 1010i)
89acc -= &m1 * &m2;
90assert_eq!(acc, (990, 1010));
91```
92
93The `Complex` number type supports various functions. Most methods have four
94versions:
95
96 1. The first method consumes the operand and rounds the returned `Complex`
97    number to the [nearest][Round::Nearest] representable value.
98 2. The second method has a “`_mut`” suffix, mutates the operand and rounds it
99    the nearest representable value.
100 3. The third method has a “`_round`” suffix, mutates the operand, applies the
101    specified [rounding method][Round] to the real and imaginary parts, and
102    returns the rounding direction for both:
103      * <code>[Ordering]::[Less][Ordering::Less]</code> if the stored part is less than the
104        exact result,
105      * <code>[Ordering]::[Equal][Ordering::Equal]</code> if the stored part is
106        equal to the exact result,
107      * <code>[Ordering]::[Greater][Ordering::Greater]</code> if the stored part
108        is greater than the exact result.
109 4. The fourth method has a “`_ref`” suffix and borrows the operand. The
110    returned item is an [incomplete-computation value][icv] that can be assigned
111    to a `Complex` number; the rounding method is selected during the
112    assignment.
113
114```rust
115use core::cmp::Ordering;
116use rug::float::Round;
117use rug::Complex;
118let expected = Complex::with_val(53, (1.2985, 0.6350));
119
120// 1. consume the operand, round to nearest
121let a = Complex::with_val(53, (1, 1));
122let sin_a = a.sin();
123assert!(*(sin_a - &expected).abs().real() < 0.0001);
124
125// 2. mutate the operand, round to nearest
126let mut b = Complex::with_val(53, (1, 1));
127b.sin_mut();
128assert!(*(b - &expected).abs().real() < 0.0001);
129
130// 3. mutate the operand, apply specified rounding
131let mut c = Complex::with_val(4, (1, 1));
132// using 4 significant bits, 1.2985 is rounded down to 1.25
133// and 0.6350 is rounded down to 0.625.
134let dir = c.sin_round((Round::Nearest, Round::Nearest));
135assert_eq!(c, (1.25, 0.625));
136assert_eq!(dir, (Ordering::Less, Ordering::Less));
137
138// 4. borrow the operand
139let d = Complex::with_val(53, (1, 1));
140let r = d.sin_ref();
141let sin_d = Complex::with_val(53, r);
142assert!(*(sin_d - &expected).abs().real() < 0.0001);
143// d was not consumed
144assert_eq!(d, (1, 1));
145```
146
147[icv]: crate#incomplete-computation-values
148*/

520    /// Creates a [`Complex`] number from an initialized [MPC complex
521    /// number][mpc_t].
522    ///
523    /// # Safety
524    ///
525    ///   * The function must *not* be used to create a constant [`Complex`]
526    ///     number, though it can be used to create a static [`Complex`] number.
527    ///     This is because constant values are *copied* on use, leading to
528    ///     undefined behavior when they are dropped.
529    ///   * The value must be initialized as a valid [`mpc_t`].
530    ///   * The [`mpc_t`] type can be considered as a kind of pointer, so there
531    ///     can be multiple copies of it. Since this function takes over
532    ///     ownership, no other copies of the passed value should exist.
533    ///
534    /// # Examples
535    ///
536    /// ```rust
537    /// use gmp_mpfr_sys::mpc;
538    /// use core::mem::MaybeUninit;
539    /// use rug::Complex;
540    /// let c = unsafe {
541    ///     let mut m = MaybeUninit::uninit();
542    ///     mpc::init3(m.as_mut_ptr(), 53, 53);
543    ///     let mut m = m.assume_init();
544    ///     mpc::set_d_d(&mut m, -14.5, 3.25, mpc::RNDNN);
545    ///     // m is initialized and unique
546    ///     Complex::from_raw(m)
547    /// };
548    /// assert_eq!(c, (-14.5, 3.25));
549    /// // since c is a Complex now, deallocation is automatic
550    /// ```
551    ///
552    /// This can be used to create a static [`Complex`] number. See [`mpc_t`],
553    /// [`mpfr_t`] and the [MPFR documentation][mpfr internals] for details.
554    ///
555    /// ```rust
556    /// use core::ptr::NonNull;
557    /// use gmp_mpfr_sys::gmp::limb_t;
558    /// use gmp_mpfr_sys::mpfr::{mpfr_t, prec_t};
559    /// use gmp_mpfr_sys::mpc::mpc_t;
560    /// use rug::{Complex, Float};
561    /// const LIMBS: [limb_t; 2] = [5, 1 << (limb_t::BITS - 1)];
562    /// const LIMBS_PTR: *const [limb_t; 2] = &LIMBS;
563    /// const MANTISSA_DIGITS: u32 = limb_t::BITS * 2;
564    /// const MPC: mpc_t = mpc_t {
565    ///     re: mpfr_t {
566    ///         prec: MANTISSA_DIGITS as prec_t,
567    ///         sign: -1,
568    ///         exp: 1,
569    ///         d: unsafe { NonNull::new_unchecked(LIMBS_PTR.cast_mut().cast()) },
570    ///     },
571    ///     im: mpfr_t {
572    ///         prec: MANTISSA_DIGITS as prec_t,
573    ///         sign: 1,
574    ///         exp: 1,
575    ///         d: unsafe { NonNull::new_unchecked(LIMBS_PTR.cast_mut().cast()) },
576    ///     },
577    /// };
578    /// // Must *not* be const, otherwise it would lead to undefined
579    /// // behavior on use, as it would create a copy that is dropped.
580    /// static C: Complex = unsafe { Complex::from_raw(MPC) };
581    /// let lsig = Float::with_val(MANTISSA_DIGITS, 5) >> (MANTISSA_DIGITS - 1);
582    /// let msig = 1u32;
583    /// let val = lsig + msig;
584    /// let check = Complex::from((-val.clone(), val));
585    /// assert_eq!(C, check);
586    /// ```
587    ///
588    /// [`mpfr_t`]: gmp_mpfr_sys::mpfr::mpfr_t
589    /// [mpfr internals]: gmp_mpfr_sys::C::MPFR::MPFR_Interface#Internals
590    #[inline]
591    pub const unsafe fn from_raw(raw: mpc_t) -> Self {
592        Complex { inner: raw }
593    }

595    /// Converts a [`Complex`] number into an [MPC complex number][mpc_t].
596    ///
597    /// The returned object should be freed to avoid memory leaks.
598    ///
599    /// # Examples
600    ///
601    /// ```rust
602    /// use gmp_mpfr_sys::mpc;
603    /// use gmp_mpfr_sys::mpfr;
604    /// use gmp_mpfr_sys::mpfr::rnd_t;
605    /// use rug::Complex;
606    /// let c = Complex::with_val(53, (-14.5, 3.25));
607    /// let mut m = c.into_raw();
608    /// unsafe {
609    ///     let re_ptr = mpc::realref_const(&m);
610    ///     let re = mpfr::get_d(re_ptr, rnd_t::RNDN);
611    ///     assert_eq!(re, -14.5);
612    ///     let im_ptr = mpc::imagref_const(&m);
613    ///     let im = mpfr::get_d(im_ptr, rnd_t::RNDN);
614    ///     assert_eq!(im, 3.25);
615    ///     // free object to prevent memory leak
616    ///     mpc::clear(&mut m);
617    /// }
618    /// ```
619    #[inline]
620    pub const fn into_raw(self) -> mpc_t {
621        let ret = self.inner;
622        let _ = ManuallyDrop::new(self);

1234    ///
1235    /// For ordering, the real part has precedence over the imaginary part.
1236    /// Negative zero is ordered as less than positive zero. Negative NaN is
1237    /// ordered as less than negative infinity, while positive NaN is ordered as
1238    /// greater than positive infinity. Comparing two negative NaNs or two
1239    /// positive NaNs produces equality.
1240    ///
1241    /// # Examples
1242    ///
1243    /// ```rust
1244    /// use rug::float::Special;

1246    /// let mut values = vec![

1248    ///     Complex::with_val(53, (Special::Zero, Special::NegZero)),
1249    ///     Complex::with_val(53, (Special::NegZero, Special::Infinity)),
1250    /// ];
1251    ///
1252    /// values.sort_by(Complex::total_cmp);
1253    ///
1254    /// // (-0, +∞)
1255    /// assert!(values[0].real().is_zero() && values[0].real().is_sign_negative());
1256    /// assert!(values[0].imag().is_infinite() && values[0].imag().is_sign_positive());
1257    /// // (+0, -0)
1258    /// assert!(values[1].real().is_zero() && values[1].real().is_sign_positive());
1259    /// assert!(values[1].imag().is_zero() && values[1].imag().is_sign_negative());
1260    /// // (+0, +0)

1265    pub fn total_cmp(&self, other: &Complex) -> Ordering {
1266        self.real()
1267            .total_cmp(other.real())
1268            .then_with(|| self.imag().total_cmp(other.imag()))
1269    }

1962        let dir_im = imag.assign_round(Special::Zero, round.1);
1963        (dir_re, dir_im)

4493fn parse(mut bytes: &[u8], radix: i32) -> Result<ParseIncomplete, ParseComplexError> {
4494    bytes = misc::trim_start(bytes);
4495    bytes = misc::trim_end(bytes);
4496    if bytes.is_empty() {
4497        return parse_error!(ParseErrorKind::NoDigits);
4498    }
4499    if let Some((inside, remainder)) = misc::matched_brackets(bytes) {
4500        if !misc::trim_start(remainder).is_empty() {
4501            return parse_error!(ParseErrorKind::CloseNotLast);
4502        }
4503        bytes = misc::trim_start(inside);
4504        bytes = misc::trim_end(bytes);
4505    } else {
4506        return match Float::parse_radix(bytes, radix) {
4507            Ok(re) => Ok(ParseIncomplete::Real(re)),
4508            Err(e) => parse_error!(ParseErrorKind::InvalidFloat(e)),
4509        };
4510    };
4511    let (real, imag) = if let Some(comma) = misc::find_outside_brackets(bytes, b',') {
4512        let real = misc::trim_end(&bytes[..comma]);
4513        if real.is_empty() {
4514            return parse_error!(ParseErrorKind::NoRealDigits);
4515        }
4516        let imag = misc::trim_start(&bytes[comma + 1..]);
4517        if imag.is_empty() {
4518            return parse_error!(ParseErrorKind::NoImagDigits);
4519        }
4520        if misc::find_outside_brackets(imag, b',').is_some() {
4521            return parse_error!(ParseErrorKind::MultipleSeparators);
4522        }
4523        (real, imag)
4524    } else if let Some(space) = misc::find_space_outside_brackets(bytes) {
4525        let real = &bytes[..space];
4526        assert!(!real.is_empty());
4527        let imag = misc::trim_start(&bytes[space + 1..]);
4528        assert!(!imag.is_empty());
4529        if misc::find_space_outside_brackets(imag).is_some() {
4530            return parse_error!(ParseErrorKind::MultipleSeparators);
4531        }
4532        (real, imag)
4533    } else {
4534        return parse_error!(ParseErrorKind::MissingSeparator);
4535    };
4536    let re = match Float::parse_radix(real, radix) {
4537        Ok(re) => re,
4538        Err(e) => return parse_error!(ParseErrorKind::InvalidRealFloat(e)),
4539    };
4540    let im = match Float::parse_radix(imag, radix) {
4541        Ok(im) => im,
4542        Err(e) => return parse_error!(ParseErrorKind::InvalidImagFloat(e)),
4543    };
4544    Ok(ParseIncomplete::Complex(re, im))
4545}
