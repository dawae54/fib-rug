32pub trait OptRational: Copy {
33    const IS_SOME: bool;
34    type Part: OptInteger;
35    fn mpq(self) -> *const mpq_t;
36    fn mpq_or(self, default: *mut mpq_t) -> *const mpq_t;
37    fn parts(self) -> (Self::Part, Self::Part);
38    fn unwrap_parts<'a>(self) -> (&'a Integer, &'a Integer)
39    where
40        Self: 'a;
41}
42
43impl OptRational for () {
44    const IS_SOME: bool = false;
45    type Part = ();
46    #[inline(always)]
47    fn mpq(self) -> *const mpq_t {
48        unreachable!("unwrapping ()");
49    }
50    #[inline(always)]
51    fn mpq_or(self, default: *mut mpq_t) -> *const mpq_t {
52        default.cast_const()
53    }
54    #[inline(always)]
55    fn parts(self) -> ((), ()) {
56        ((), ())
57    }
58    #[inline(always)]
59    fn unwrap_parts<'a>(self) -> (&'a Integer, &'a Integer) {
60        unreachable!("unwrapping ()");
61    }
62}
63
64impl<'a> OptRational for &'a Rational {
65    const IS_SOME: bool = true;
66    type Part = &'a Integer;
67    #[inline(always)]
68    fn mpq(self) -> *const mpq_t {
69        self.as_raw()
70    }
71    #[inline(always)]
72    fn mpq_or(self, _default: *mut mpq_t) -> *const mpq_t {
73        self.as_raw()
74    }
75    #[inline(always)]
76    fn parts(self) -> (&'a Integer, &'a Integer) {
77        (self.numer(), self.denom())
78    }
79    #[inline(always)]
80    fn unwrap_parts<'b>(self) -> (&'b Integer, &'b Integer)
81    where
82        Self: 'b,
83    {
84        (self.numer(), self.denom())
85    }
86}

88macro_rules! unsafe_wrap {
89    (fn $fn:ident($($op:ident: $O:ident),* $(; $param:ident: $T:ty)*) -> $deleg:path) => {
90        #[inline]
91        pub fn $fn<$($O: OptRational),*>(rop: &mut Rational $(, $op: $O)* $(, $param: $T)*) {
92            let rop = rop.as_raw_mut();
93            $(let $op = $op.mpq_or(rop);)*
94            unsafe {
95                $deleg(rop $(, $op)* $(, $param.into())*);
96            }
97        }
98    };
99}

142macro_rules! int_rat {
143    (fn $fn:ident, $fn_int:ident, |$rop:ident, $num:ident, $den:ident| $body:block) => {
144        #[inline]
145        pub fn $fn(rat: &mut Rational) {
146            let (num, den) = unsafe { rat.as_mut_numer_denom_no_canonicalization() };
147            let $rop: &mut Integer = num;
148            let $num = ();
149            let $den: &Integer = &*den;
150            $body
151            xmpz::set_1(den);
152        }
153
154        #[inline]
155        pub fn $fn_int(rop: &mut Integer, op: &Rational) {
156            let $rop: &mut Integer = rop;
157            let $num = op.numer();
158            let $den: &Integer = op.denom();
159            $body
160        }
161    };
162}

164int_rat! { fn signum, signum_int, |rop, num, _den| {
165    xmpz::signum(rop, num);
166} }

168int_rat! { fn trunc, trunc_int, |rop, num, den| {
169    xmpz::tdiv_q(rop, num, den);
170} }
171
172int_rat! { fn ceil, ceil_int, |rop, num, den| {
173    // use tdiv_q rather than cdiv_q to let GMP not keep remainder
174    if xmpz::is_1(den) {
175        xmpz::set(rop, num);
176    } else {
177        let neg = unsafe { gmp::mpz_sgn(num.mpz_or(rop.as_raw_mut())) < 0 };
178        xmpz::tdiv_q(rop, num, den);
179        if !neg {
180            xmpz::add_ui(rop, (), 1);
181        }
182    }
183} }
184
185int_rat! { fn floor, floor_int, |rop, num, den| {
186    // use tdiv_q rather than fdiv_q to let GMP not keep remainder
187    if xmpz::is_1(den) {
188        xmpz::set(rop, num);
189    } else {
190        let neg = unsafe { gmp::mpz_sgn(num.mpz_or(rop.as_raw_mut())) < 0 };
191        xmpz::tdiv_q(rop, num, den);
192        if neg {
193            xmpz::sub_ui(rop, (), 1);
194        }
195    }
196} }
197
198int_rat! { fn round, round_int, |rop, num, den| {
199    // The remainder cannot be larger than the divisor, but we
200    // allocate an extra limb because the GMP docs suggest we should.
201    let limbs = den.inner().size.abs().unwrapped_as::<usize>() + 1;
202    let bits = limbs
203        .checked_mul(gmp::LIMB_BITS.az::<usize>())
204        .expect("overflow");
205    let mut rem = Integer::with_capacity(bits);
206    xmpz::tdiv_qr(rop, &mut rem, num, den);
207    if xmpz::round_away(&rem, den) {
208        if rem.is_negative() {
209            // negative number
210            xmpz::sub_ui(rop, (), 1);
211        } else {
212            // positive number
213            xmpz::add_ui(rop, (), 1);
214        }
215    }
216} }

218#[inline]
219pub fn inv<O: OptRational>(rop: &mut Rational, op: O) {
220    let rop = rop.as_raw_mut();
221    let op = op.mpq_or(rop);
222    unsafe {
223        assert_ne!(gmp::mpq_sgn(op), 0, "division by zero");
224        gmp::mpq_inv(rop, op);
225    }

282pub fn round_fract<O: OptRational>(fract: &mut Rational, op: O) {
283    let (fract_num, fract_den) = unsafe { fract.as_mut_numer_denom_no_canonicalization() };
284    let (op_num, op_den) = op.parts();
285    xmpz::set(fract_den, op_den);
286    xmpz::tdiv_r(fract_num, op_num, &*fract_den);
287    if xmpz::round_away(fract_num, fract_den) {
288        if fract_num.is_negative() {
289            // negative number
290            xmpz::add(fract_num, (), &*fract_den);
291        } else {
292            // positive number
293            xmpz::sub(fract_num, (), &*fract_den);
294        }
295    }
296}

298pub fn round_fract_whole<O: OptRational>(fract: &mut Rational, round: &mut Integer, op: O) {
299    let (fract_num, fract_den) = unsafe { fract.as_mut_numer_denom_no_canonicalization() };
300    let (op_num, op_den) = op.parts();
301    xmpz::set(fract_den, op_den);
302    let fract = fract_num.as_raw_mut();
303    let op = op_num.mpz_or(fract);
304    unsafe {
305        gmp::mpz_tdiv_qr(round.as_raw_mut(), fract, op, fract_den.as_raw());
306    }
307    if xmpz::round_away(fract_num, fract_den) {
308        if fract_num.is_negative() {
309            // negative number
310            xmpz::sub_ui(round, (), 1);
311            xmpz::add(fract_num, (), &*fract_den);
312        } else {
313            // positive number
314            xmpz::add_ui(round, (), 1);
315            xmpz::sub(fract_num, (), &*fract_den);
316        }
317    }
318}

330#[inline]
331fn sgn_or<O: OptRational>(op: O, or: &mut Rational) -> Ordering {
332    let q = op.mpq_or(or.as_raw_mut());
333    unsafe {
334        let z = gmp::mpq_numref_const(q);
335        gmp::mpz_sgn(z).cmp(&0)
336    }
337}
338
339#[inline]
340fn check_div0_or<O: OptRational>(divisor: O, or: &mut Rational) {
341    assert!(sgn_or(divisor, or) != Ordering::Equal, "division by zero");
342}

344#[inline]
345pub fn div<O: OptRational, P: OptRational>(rop: &mut Rational, op1: O, op2: P) {
346    check_div0_or(op2, rop);
347    let rop = rop.as_raw_mut();
348    let op1 = op1.mpq_or(rop);
349    let op2 = op2.mpq_or(rop);
350    unsafe {
351        gmp::mpq_div(rop, op1, op2);
352    }
353}

355// num and den must form a canonical pair
356#[inline]
357pub unsafe fn write_num_den_unchecked(dst: &mut MaybeUninit<Rational>, num: Integer, den: Integer) {
358    let inner_ptr = misc::cast_ptr_mut(dst.as_mut_ptr());
359    unsafe {
360        let num_ptr = misc::cast_ptr_mut::<_, Integer>(gmp::mpq_numref(inner_ptr));
361        num_ptr.write(num);
362        let den_ptr = misc::cast_ptr_mut::<_, Integer>(gmp::mpq_denref(inner_ptr));
363        den_ptr.write(den);
364    }
365}

367#[inline]
368pub fn write_num_den_canonicalize(dst: &mut MaybeUninit<Rational>, num: Integer, den: Integer) {
369    assert!(!den.is_zero(), "division by zero");
370    // Safety:
371    //   * We can cast pointers to/from Rational/mpq_t as they are repr(transparent).
372    //   * We can cast pointers to/from Integer/mpz_t as they are repr(transparent).
373    //   * numref/denref only offset the pointers, and can operate on uninit memory.
374    unsafe {
375        let inner_ptr = misc::cast_ptr_mut(dst.as_mut_ptr());
376        let num_ptr = misc::cast_ptr_mut::<_, Integer>(gmp::mpq_numref(inner_ptr));
377        num_ptr.write(num);
378        let den_ptr = misc::cast_ptr_mut::<_, Integer>(gmp::mpq_denref(inner_ptr));
379        den_ptr.write(den);
380        gmp::mpq_canonicalize(inner_ptr);
381    }
382}

384#[inline]
385pub fn canonicalize(r: &mut Rational) {
386    if r.denom().is_zero() {
387        // Leave in canonical state before panic (issue 49).
388        unsafe {
389            let den = numref_denref(r).1;
390            xmpz::set_1(den);
391        }
392        panic!("division by zero");
393    }
394    unsafe {
395        gmp::mpq_canonicalize(r.as_raw_mut());
396    }
397}

535#[inline]
536pub fn cmp_u64(op1: &Rational, n2: u64, d2: u64) -> Ordering {
537    if let Some(n2) = n2.checked_as() {
538        if let Some(d2) = d2.checked_as() {
539            return ord(unsafe { gmp::mpq_cmp_ui(op1.as_raw(), n2, d2) });
540        }
541    }
542    let mut small = MiniRational::from((n2, d2));
543    ord(unsafe { gmp::mpq_cmp(op1.as_raw(), small.borrow_excl().as_raw()) })
544}

569pub fn cmp_i128(op1: &Rational, n2: i128, d2: u128) -> Ordering {
570    if let Some(n2) = n2.checked_as() {
571        if let Some(d2) = d2.checked_as() {
572            return ord(unsafe { gmp::mpq_cmp_si(op1.as_raw(), n2, d2) });
573        }
574    }
575    let mut small = MiniRational::from((n2, d2));
576    ord(unsafe { gmp::mpq_cmp(op1.as_raw(), small.borrow_excl().as_raw()) })
577}

579pub fn cmp_finite_d(op1: &Rational, op2: f64) -> Ordering {
580    let num1 = op1.numer();
581    let den1 = op1.denom();
582    let den1_bits = den1.significant_bits();
583    // cmp(num1, op2 * den1)
584    let cmp;
585    unsafe {
586        let mut op2_f = MaybeUninit::uninit();
587        gmp::mpf_init2(op2_f.as_mut_ptr(), 53);
588        let mut op2_f = op2_f.assume_init();
589        gmp::mpf_set_d(&mut op2_f, op2);
590        let mut rhs = MaybeUninit::uninit();
591        gmp::mpf_init2(rhs.as_mut_ptr(), (den1_bits + 53).unwrapped_cast());
592        let mut rhs = rhs.assume_init();
593        gmp::mpf_set_z(&mut rhs, den1.as_raw());
594        gmp::mpf_mul(&mut rhs, &rhs, &op2_f);
595        cmp = -gmp::mpf_cmp_z(&rhs, num1.as_raw());
596        gmp::mpf_clear(&mut rhs);
597        gmp::mpf_clear(&mut op2_f);
598    }
599    ord(cmp)
600}

626pub fn mul_z<O: OptRational>(rop: &mut Rational, lhs: O, rhs: &Integer) {
627    if rhs.is_zero() {
628        set_0(rop);
629        return;
630    }
631    // Canonicalization is done in this function.
632    //     gcd = gcd(lhs.denom, rhs)
633    //     (numer, denom) = (rhs / gcd * lhs.numer, lhs.denom / gcd)
634    let (numer, denom) = unsafe { rop.as_mut_numer_denom_no_canonicalization() };
635    if O::IS_SOME {
636        let (lhs_num, lhs_den) = lhs.unwrap_parts();
637        // store gcd temporarily in numer
638        numer.assign(lhs_den.gcd_ref(rhs));
639        if xmpz::is_1(numer) {
640            numer.assign(lhs_num * rhs);
641            denom.assign(lhs_den);
642        } else {
643            denom.assign(lhs_den.div_exact_ref(numer));
644            numer.div_exact_from(rhs);
645            *numer *= lhs_num;
646        }
647    } else {
648        let mut gcd = Integer::from(denom.gcd_ref(rhs));
649        if xmpz::is_1(&gcd) {
650            *numer *= rhs;
651        } else {
652            denom.div_exact_mut(&gcd);
653            gcd.div_exact_from(rhs);
654            *numer *= gcd;
655        }
656    }
657}

659pub fn div_z<O: OptRational>(rop: &mut Rational, lhs: O, rhs: &Integer) {
660    xmpz::check_div0(rhs);
661    // Canonicalization is done in this function.
662    //     gcd = gcd(lhs.numer, rhs)
663    //     (numer, denom) = (lhs.numer / gcd, rhs / gcd * lhs.denom)
664    let (numer, denom) = unsafe { rop.as_mut_numer_denom_no_canonicalization() };
665    if O::IS_SOME {
666        let (lhs_num, lhs_den) = lhs.unwrap_parts();
667        // store gcd temporarily in numer
668        numer.assign(lhs_num.gcd_ref(rhs));
669        if xmpz::is_1(numer) {
670            numer.assign(lhs_num);
671            denom.assign(lhs_den * rhs);
672        } else {
673            denom.assign(rhs.div_exact_ref(numer));
674            *denom *= lhs_den;
675            numer.div_exact_from(lhs_num);
676        }
677    } else {
678        let mut gcd = Integer::from(numer.gcd_ref(rhs));
679        if xmpz::is_1(&gcd) {
680            *denom *= rhs;
681        } else {
682            numer.div_exact_mut(&gcd);
683            gcd.div_exact_from(rhs);
684            *denom *= gcd;
685        }
686    }
687    if denom.is_negative() {
688        numer.neg_assign();
689        denom.neg_assign();
690    }
691}

693pub fn z_div<O: OptRational>(rop: &mut Rational, lhs: &Integer, rhs: O) {
694    if O::IS_SOME {
695        xmpz::check_div0(rhs.unwrap_parts().0);
696    } else {
697        xmpz::check_div0(rop.numer());
698    }
699    // Canonicalization is done in this function.
700    //     gcd = gcd(lhs, rhs.numer)
701    //     (numer, denom) = (lhs / gcd * rhs.denom, rhs.numer / gcd)
702    let (numer, denom) = unsafe { rop.as_mut_numer_denom_no_canonicalization() };
703    if O::IS_SOME {
704        let (rhs_num, rhs_den) = rhs.unwrap_parts();
705        // store gcd temporarily in numer
706        numer.assign(rhs_num.gcd_ref(lhs));
707        if xmpz::is_1(numer) {
708            numer.assign(lhs * rhs_den);
709            denom.assign(rhs_num);
710        } else {
711            denom.assign(rhs_num.div_exact_ref(numer));
712            numer.div_exact_from(lhs);
713            *numer *= rhs_den;
714        }
715    } else {
716        let mut gcd = Integer::from(numer.gcd_ref(lhs));
717        mem::swap(numer, denom);
718        if xmpz::is_1(&gcd) {
719            *numer *= lhs;
720        } else {
721            denom.div_exact_mut(&gcd);
722            gcd.div_exact_from(lhs);
723            *numer *= gcd;
724        }
725    }
726    if denom.is_negative() {
727        numer.neg_assign();
728        denom.neg_assign();
729    }
730}

793pub fn mul_ui<O: OptRational>(rop: &mut Rational, lhs: O, rhs: c_ulong) {
794    if rhs == 0 {
795        set_0(rop);
796        return;
797    }
798    // Canonicalization is done in this function.
799    //     gcd = gcd(lhs.denom, rhs), cannot be zero because rhs ≠ 0
800    //     (numer, denom) = (rhs / gcd * lhs.numer, lhs.denom / gcd)
801    let (numer, denom) = unsafe { rop.as_mut_numer_denom_no_canonicalization() };
802    if O::IS_SOME {
803        let (lhs_num, lhs_den) = lhs.unwrap_parts();
804        let gcd = xmpz::gcd_opt_ui(None, lhs_den, rhs);
805        if gcd == 1 {
806            numer.assign(rhs * lhs_num);
807            denom.assign(lhs_den);
808        } else {
809            numer.assign(rhs / gcd * lhs_num);
810            xmpz::divexact_ui(denom, lhs_den, gcd);
811        }
812    } else {
813        let gcd = xmpz::gcd_opt_ui(None, &*denom, rhs);
814        if gcd == 1 {
815            *numer *= rhs;
816        } else {
817            *numer *= rhs / gcd;
818            xmpz::divexact_ui(denom, (), gcd);
819        }
820    }
821}

823pub fn div_ui<O: OptRational>(rop: &mut Rational, lhs: O, rhs: c_ulong) {
824    assert_ne!(rhs, 0, "division by zero");
825    // Canonicalization is done in this function.
826    //     gcd = gcd(lhs.numer, rhs), cannot be zero because rhs ≠ 0
827    //     (numer, denom) = (lhs.numer / gcd, rhs / gcd * lhs.denom)
828    let (numer, denom) = unsafe { rop.as_mut_numer_denom_no_canonicalization() };
829    if O::IS_SOME {
830        let (lhs_num, lhs_den) = lhs.unwrap_parts();
831        let gcd = xmpz::gcd_opt_ui(None, lhs_num, rhs);
832        if gcd == 1 {
833            numer.assign(lhs_num);
834            denom.assign(rhs * lhs_den);
835        } else {
836            xmpz::divexact_ui(numer, lhs_num, gcd);
837            denom.assign(rhs / gcd * lhs_den);
838        }
839    } else {
840        let gcd = xmpz::gcd_opt_ui(None, &*numer, rhs);
841        if gcd == 1 {
842            *denom *= rhs;
843        } else {
844            xmpz::divexact_ui(numer, (), gcd);
845            *denom *= rhs / gcd;
846        }
847    }
848    // since rhs is positive, denom is positive
849}

851pub fn ui_div<O: OptRational>(rop: &mut Rational, lhs: c_ulong, rhs: O) {
852    if O::IS_SOME {
853        xmpz::check_div0(rhs.unwrap_parts().0);
854    } else {
855        xmpz::check_div0(rop.numer());
856    }
857    if lhs == 0 {
858        set_0(rop);
859        return;
860    }
861    // Canonicalization is done in this function.
862    //     gcd = gcd(lhs, rhs.numer), cannot be zero because lhs ≠ 0
863    //     (numer, denom) = (lhs / gcd * rhs.denom, rhs.numer / gcd)
864    let (numer, denom) = unsafe { rop.as_mut_numer_denom_no_canonicalization() };
865    if O::IS_SOME {
866        let (rhs_num, rhs_den) = rhs.unwrap_parts();
867        let gcd = xmpz::gcd_opt_ui(None, rhs_num, lhs);
868        if gcd == 1 {
869            numer.assign(lhs * rhs_den);
870            denom.assign(rhs_num);
871        } else {
872            numer.assign(lhs / gcd * rhs_den);
873            xmpz::divexact_ui(denom, rhs_num, gcd);
874        }
875    } else {
876        let gcd = xmpz::gcd_opt_ui(None, &*numer, lhs);
877        mem::swap(numer, denom);
878        if gcd == 1 {
879            *numer *= lhs;
880        } else {
881            *numer *= lhs / gcd;
882            xmpz::divexact_ui(denom, (), gcd);
883        }
884    }
885    if denom.is_negative() {
886        numer.neg_assign();
887        denom.neg_assign();
888    }
889}
