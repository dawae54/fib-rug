28/// Used to get a reference to an [`Integer`].
29///
30/// The struct implements <code>[Deref]\<[Target][Deref::Target] = [Integer]></code>.
31///
32/// No memory is unallocated when this struct is dropped.
33///
34/// # Examples
35///
36/// ```rust
37/// use rug::integer::BorrowInteger;
38/// use rug::Integer;
39/// let i = Integer::from(42);
40/// let neg: BorrowInteger = i.as_neg();
41/// // i is still valid
42/// assert_eq!(i, 42);
43/// assert_eq!(*neg, -42);
44/// ```
45#[derive(Clone, Copy)]
46#[repr(transparent)]
47pub struct BorrowInteger<'a> {
48    inner: mpz_t,
49    phantom: PhantomData<&'a Integer>,
50}
51
52impl<'a> BorrowInteger<'a> {
53    /// Create a borrow from a raw [GMP integer][mpz_t].
54    ///
55    /// # Safety
56    ///
57    ///   * The value must be initialized as a valid [`mpz_t`].
58    ///   * The [`mpz_t`] type can be considered as a kind of pointer, so there
59    ///     can be multiple copies of it. [`BorrowInteger`] cannot mutate the
60    ///     value, so there can be other copies, but none of them are allowed to
61    ///     mutate the value.
62    ///   * The lifetime is obtained from the return type. The user must ensure
63    ///     the value remains valid for the duration of the lifetime.
64    ///
65    /// # Examples
66    ///
67    /// ```rust
68    /// use rug::integer::BorrowInteger;
69    /// use rug::Integer;
70    /// let i = Integer::from(42);
71    /// // Safety: i.as_raw() is a valid pointer.
72    /// let raw = unsafe { *i.as_raw() };
73    /// // Safety: i is still valid when borrow is used.
74    /// let borrow = unsafe { BorrowInteger::from_raw(raw) };
75    /// assert_eq!(i, *borrow);
76    /// ```
77    #[inline]
78    pub const unsafe fn from_raw(raw: mpz_t) -> BorrowInteger<'a> {
79        BorrowInteger {
80            inner: raw,
81            phantom: PhantomData,
82        }
83    }
84
85    /// Gets a reference to [`Integer`] from a `BorrowInteger`.
86    ///
87    /// This is equivalent to taking the reference of the dereferencing operator
88    /// `*` or to <code>[Deref]::[deref][Deref::deref]</code>, but can also be
89    /// used in constant context. Unless required in constant context, use the
90    /// operator or trait instead.
91    ///
92    /// # Planned deprecation
93    ///
94    /// This method will be deprecated when the unary `*` operator and the
95    /// [`Deref`] trait are usable in constant context.
96    ///
97    /// # Examples
98    ///
99    /// ```rust
100    /// use core::ops::Deref;
101    /// use core::ptr;
102    /// use rug::integer::BorrowInteger;
103    /// use rug::Integer;
104    ///
105    /// let i = Integer::from(23);
106    /// let b = i.as_neg();
107    /// let using_method: &Integer = BorrowInteger::const_deref(&b);
108    /// let using_operator: &Integer = &*b;
109    /// let using_trait: &Integer = b.deref();
110    /// assert_eq!(*using_method, -23);
111    /// assert!(ptr::eq(using_method, using_operator));
112    /// assert!(ptr::eq(using_method, using_trait));
113    /// ```
114    ///
115    /// This method can be used to create a constant reference.
116    ///
117    /// ```rust
118    /// use rug::integer::{BorrowInteger, MiniInteger};
119    /// use rug::Integer;
120    ///
121    /// const TWO_MINI: MiniInteger = MiniInteger::const_from_i32(2);
122    /// const TWO_BORROW: BorrowInteger = TWO_MINI.borrow();
123    /// const TWO: &Integer = BorrowInteger::const_deref(&TWO_BORROW);
124    /// assert_eq!(*TWO, 2);
125    /// ```
126    ///
127    /// If the constant cannot be initialized using
128    /// [`MiniInteger`][crate::integer::MiniInteger], we can set up the limbs
129    /// directly, though it is more complicated.
130    ///
131    /// ```rust
132    /// use gmp_mpfr_sys::gmp;
133    /// use gmp_mpfr_sys::gmp::{limb_t, mpz_t};
134    /// use rug::integer::BorrowInteger;
135    /// use rug::Integer;
136    ///
137    /// const LIMBS: [limb_t; 2] = [123, 456];
138    /// const MPZ: mpz_t =
139    ///     unsafe { gmp::MPZ_ROINIT_N(LIMBS.as_ptr().cast_mut(), -2) };
140    /// // Safety: MPZ will remain valid, and will not be changed.
141    /// const BORROW: BorrowInteger = unsafe { BorrowInteger::from_raw(MPZ) };
142    /// const I: &Integer = BorrowInteger::const_deref(&BORROW);
143    /// let check = -((Integer::from(LIMBS[1]) << gmp::NUMB_BITS) + LIMBS[0]);
144    /// assert_eq!(*I, check);
145    /// ```
146    #[inline]
147    pub const fn const_deref<'b>(b: &'b BorrowInteger<'a>) -> &'b Integer {
148        let ptr = misc::cast_ptr(&b.inner);
149        // Safety: the inner pointer is valid for the duration of the lifetime.
150        unsafe { &*ptr }
151    }
152}
153
154impl Deref for BorrowInteger<'_> {
155    type Target = Integer;
156    #[inline]
157    fn deref(&self) -> &Integer {
158        BorrowInteger::const_deref(self)
159    }
160}
