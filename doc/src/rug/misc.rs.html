30pub trait NegAbs {
31    type Abs;
32    fn neg_abs(self) -> (bool, Self::Abs);
33}
34
35macro_rules! neg_abs {
36    ($I:ty; $U:ty) => {
37        impl NegAbs for $I {
38            type Abs = $U;
39            #[inline]
40            fn neg_abs(self) -> (bool, $U) {
41                if self < 0 {
42                    (true, self.wrapping_neg().wrapping_cast())
43                } else {
44                    (false, self.wrapping_cast())
45                }
46            }
47        }
48
49        impl NegAbs for $U {
50            type Abs = $U;
51            #[inline]
52            fn neg_abs(self) -> (bool, $U) {
53                (false, self)
54            }
55        }
56    };
57}

66pub fn trunc_f64_to_f32(f: f64) -> f32 {
67    // f as f32 might round away from zero, so we need to clear
68    // the least significant bits of f.
69    //   * If f is a nan, we do NOT want to clear any mantissa bits,
70    //     as this may change f into +/- infinity.
71    //   * If f is +/- infinity, the bits are already zero, so the
72    //     masking has no effect.
73    //   * If f is subnormal, f as f32 will be zero anyway.
74    //
75    // When f is normal but would be subnormal as f32, we need to clear more
76    // bits. Let x be exponent minus minimum f32 normal exponent, that is x =
77    // biased f64 exponent - 1023 + 126. Then
78    //   * If x >= 0, then truncate 53 - 24 bits.
79    //   * If x <= -24, then truncate at least 53 bits, but there are 52
80    //     non-implicit bits, so return 0.
81    //   * If -23 <= x <= -1, then truncate 53 - 24 - x bits.
82    if f.is_nan() {
83        f as f32
84    } else {
85        let u = f.to_bits();
86        let biased_exp = (u >> 52).az::<u32>() & 0x7FF;
87        let trunc_count = if biased_exp >= 1023 - 126 {
88            // normally f64 has 29 more significant bits than f32
89            29
90        } else if biased_exp <= 1023 - 126 - 24 {
91            // Do not try to keep sign bit, as that is not consistent with
92            // gmp::mpz_get_d
93            return 0.0;
94        } else {
95            // 1023 - 126 - 23 <= biased_exp <= 1023 - 126 - 1
96            // 52 >= trunc_count >= 30
97            29 + 1023 - 126 - biased_exp
98        };
99        // f64 has 29 more significant bits than f32.
100        let trunc_u = u & (!0 << trunc_count);
101        let trunc_f = f64::from_bits(trunc_u);
102        trunc_f as f32
103    }
104}

106fn lcase(byte: u8) -> u8 {
107    match byte {
108        b'A'..=b'Z' => byte - b'A' + b'a',
109        _ => byte,
110    }
111}

113pub fn trim_start(bytes: &[u8]) -> &[u8] {
114    for (start, &b) in bytes.iter().enumerate() {
115        match b {
116            b' ' | b'\t' | b'\n' | 0x0b | 0x0c | 0x0d => {}
117            _ => return &bytes[start..],
118        }
119    }
120    &[]
121}
122
123pub fn trim_end(bytes: &[u8]) -> &[u8] {
124    for (end, &b) in bytes.iter().enumerate().rev() {
125        match b {
126            b' ' | b'\t' | b'\n' | 0x0b | 0x0c | 0x0d => {}
127            _ => return &bytes[..=end],
128        }
129    }
130    &[]
131}

133// If bytes starts with a match to one of patterns, return bytes with
134// the match skipped. Only bytes is converted to lcase.
135pub fn skip_lcase_match<'a>(bytes: &'a [u8], patterns: &[&[u8]]) -> Option<&'a [u8]> {
136    'next_pattern: for pattern in patterns {
137        if bytes.len() < pattern.len() {
138            continue 'next_pattern;
139        }
140        for (&b, &p) in bytes.iter().zip(pattern.iter()) {
141            if lcase(b) != p {
142                continue 'next_pattern;
143            }
144        }
145        return Some(&bytes[pattern.len()..]);
146    }
147    None
148}

150// If bytes starts with '(' and has a matching ')', returns the
151// contents and the remainder.
152pub fn matched_brackets(bytes: &[u8]) -> Option<(&[u8], &[u8])> {
153    let mut iter = bytes.iter().enumerate();
154    match iter.next() {
155        Some((_, &b'(')) => {}
156        _ => return None,
157    }
158    let mut level = 1;
159    for (i, &b) in iter {
160        match b {
161            b'(' => level += 1,
162            b')' => {
163                level -= 1;
164                if level == 0 {
165                    return Some((&bytes[1..i], &bytes[i + 1..]));
166                }
167            }
168            _ => {}
169        }
170    }
171    None
172}

174pub fn find_outside_brackets(bytes: &[u8], pattern: u8) -> Option<usize> {
175    let mut level = 0;
176    for (i, &b) in bytes.iter().enumerate() {
177        match b {
178            b'(' => level += 1,
179            b')' if level > 0 => level -= 1,
180            _ if level == 0 && b == pattern => return Some(i),
181            _ => {}
182        }
183    }
184    None
185}
186
187pub fn find_space_outside_brackets(bytes: &[u8]) -> Option<usize> {
188    let mut level = 0;
189    for (i, &b) in bytes.iter().enumerate() {
190        match b {
191            b'(' => level += 1,
192            b')' if level > 0 => level -= 1,
193            b' ' | b'\t' | b'\n' | 0x0b | 0x0c | 0x0d if level == 0 => {
194                return Some(i);
195            }
196            _ => {}
197        }
198    }
199    None
200}

202pub enum StringLike {
203    #[cfg(feature = "std")]
204    String(String),
205    Malloc {
206        ptr: *mut c_char,
207        cap: size_t,
208        len: size_t,
209    },
210}
211
212impl StringLike {
213    #[cfg(feature = "std")]
214    pub fn new_string() -> Self {
215        StringLike::String(String::new())
216    }
217
218    #[cfg(feature = "std")]
219    pub fn unwrap_string(mut self) -> String {
220        match &mut self {
221            StringLike::String(s) => mem::take(s),
222            StringLike::Malloc { .. } => unreachable!("unexpected variant"),
223        }
224    }
225
226    pub fn new_malloc() -> Self {
227        StringLike::Malloc {
228            ptr: NonNull::dangling().as_ptr(),
229            cap: 0,
230            len: 0,
231        }
232    }
233
234    pub fn push_str(&mut self, s: &str) {
235        #[cfg(feature = "std")]
236        if let StringLike::String(st) = self {
237            st.push_str(s);
238            return;
239        }
240        self.reserve(s.len());
241        #[cfg(feature = "std")]
242        let StringLike::Malloc { ptr, cap: _, len } = self
243        else {
244            unreachable!();
245        };
246        #[cfg(not(feature = "std"))]
247        let StringLike::Malloc { ptr, cap: _, len } = self;
248
249        // SAFETY: ptr is non-null and properly aligned, as it is either a
250        // dangling pointer created with NonNull::dangling, or a pointer
251        // returned by malloc/realloc with non-zero size.
252        unsafe {
253            ptr.cast::<u8>()
254                .offset((*len).unwrapped_as())
255                .copy_from_nonoverlapping(s.as_ptr(), s.len());
256            self.increase_len(s.len());
257        }
258    }
259
260    #[inline]
261    pub fn as_str(&self) -> &str {
262        match self {
263            #[cfg(feature = "std")]
264            StringLike::String(s) => s.as_str(),
265            StringLike::Malloc { ptr, cap: _, len } => {
266                // SAFETY: ptr is non-null and properly aligned, as it is either
267                // a dangling pointer created with NonNull::dangling, or a
268                // pointer returned by malloc/realloc with non-zero size.
269                unsafe {
270                    let s = slice::from_raw_parts(ptr.cast::<u8>(), (*len).unwrapped_as());
271                    str::from_utf8_unchecked(s)
272                }
273            }
274        }
275    }
276
277    #[inline]
278    pub fn as_mut_str(&mut self) -> &mut str {
279        match self {
280            #[cfg(feature = "std")]
281            StringLike::String(s) => s.as_mut_str(),
282            StringLike::Malloc { ptr, cap: _, len } => {
283                // SAFETY: ptr is non-null and properly aligned, as it is either
284                // a dangling pointer created with NonNull::dangling, or a
285                // pointer returned by malloc/realloc with non-zero size.
286                unsafe {
287                    let s = slice::from_raw_parts_mut(ptr.cast::<u8>(), (*len).unwrapped_as());
288                    str::from_utf8_unchecked_mut(s)
289                }
290            }
291        }
292    }
293
294    pub fn reserve(&mut self, additional: usize) {
295        match self {
296            #[cfg(feature = "std")]
297            StringLike::String(s) => {
298                s.reserve(additional);
299            }
300            StringLike::Malloc { ptr, cap, len } => {
301                let new_cap = len
302                    .checked_add(additional.unwrapped_as::<size_t>())
303                    .expect("overflow");
304                if new_cap > *cap {
305                    let new_ptr = if *cap == 0 {
306                        // cannot use realloc, as for empty slice ptr is
307                        // dangling to satisfy slice::from_raw_parts_mut, which
308                        // does not allow null pointer
309                        unsafe { libc::malloc(new_cap) }
310                    } else {
311                        unsafe { libc::realloc(ptr.cast(), new_cap) }
312                    };
313                    *ptr = new_ptr.cast::<c_char>();
314                    *cap = new_cap;
315                }
316            }
317        }
318    }
319
320    pub fn reserved_space(&mut self) -> &mut [MaybeUninit<u8>] {
321        match self {
322            #[cfg(feature = "std")]
323            StringLike::String(s) => {
324                let mu_ptr = s.as_mut_ptr().cast::<MaybeUninit<u8>>();
325                // SAFETY: mu_ptr is non-null and properly aligned, as it is
326                // obtained from String::as_mut_ptr.
327                unsafe {
328                    slice::from_raw_parts_mut(
329                        mu_ptr.offset(s.len().unwrapped_as()),
330                        s.capacity() - s.len(),
331                    )
332                }
333            }
334            StringLike::Malloc { ptr, cap, len } => {
335                let mu_ptr = (*ptr).cast::<MaybeUninit<u8>>();
336                // SAFETY: mu_ptr is non-null and properly aligned, as it is
337                // either a dangling pointer created with NonNull::dangling, or
338                // a pointer returned by malloc/realloc with non-zero size.
339                unsafe {
340                    slice::from_raw_parts_mut(
341                        mu_ptr.offset((*len).unwrapped_as()),
342                        (*cap - *len).unwrapped_as(),
343                    )
344                }
345            }
346        }
347    }
348
349    // SAFETY: there should be enough capacity to increase length by increment,
350    // and increased length must be valid utf8
351    pub unsafe fn increase_len(&mut self, increment: usize) {
352        match self {
353            #[cfg(feature = "std")]
354            StringLike::String(s) => {
355                // SAFETY: ensured by user.
356                unsafe {
357                    let new_len = s.len().checked_add(increment).expect("overflow");
358                    s.as_mut_vec().set_len(new_len);
359                }
360            }
361            StringLike::Malloc {
362                ptr: _,
363                cap: _,
364                len,
365            } => {
366                let new_len = len
367                    .checked_add(increment.unwrapped_as::<size_t>())
368                    .expect("overflow");
369                *len = new_len;
370            }
371        }
372    }
373}
374
375impl Drop for StringLike {
376    fn drop(&mut self) {
377        match self {
378            #[cfg(feature = "std")]
379            StringLike::String(_) => {}
380            StringLike::Malloc { ptr, cap, .. } => {
381                // for zero capacity, ptr is dangling and must not be freed
382                if *cap != 0 {
383                    unsafe {
384                        libc::free(ptr.cast());
385                    }
386                }
387            }
388        }
389    }
390}
391
392impl Write for StringLike {
393    fn write_str(&mut self, s: &str) -> core::fmt::Result {
394        self.push_str(s);
395        Ok(())
396    }
397}

399#[derive(Debug)]
400pub struct VecLike<T> {
401    ptr: *mut T,
402    cap: size_t,
403    len: size_t,
404}
405
406impl<T> VecLike<T> {
407    pub fn new() -> Self {
408        VecLike {
409            ptr: NonNull::dangling().as_ptr(),
410            cap: 0,
411            len: 0,
412        }
413    }
414
415    #[inline]
416    pub fn as_slice(&self) -> &[T] {
417        // SAFETY: self.ptr is non-null and properly aligned, as it is either a
418        // dangling pointer created with NonNull::dangling, or a pointer
419        // returned by malloc/realloc with non-zero size.
420        unsafe { slice::from_raw_parts(self.ptr, self.len) }
421    }
422
423    #[inline]
424    pub fn as_mut_slice(&mut self) -> &mut [T] {
425        // SAFETY: self.ptr is non-null and properly aligned, as it is either a
426        // dangling pointer created with NonNull::dangling, or a pointer
427        // returned by malloc/realloc with non-zero size.
428        unsafe { slice::from_raw_parts_mut(self.ptr, self.len) }
429    }
430
431    pub fn reserve(&mut self, additional: usize) {
432        let new_cap = self.len.checked_add(additional).expect("overflow");
433        if new_cap > self.cap {
434            let bytes_cap = new_cap
435                .checked_mul(mem::size_of::<T>())
436                .expect("overflow")
437                .unwrapped_as();
438            let new_ptr = if self.cap == 0 {
439                // cannot use realloc, as for empty slice self.ptr is dangling
440                // to satisfy slice::from_raw_parts_mut, which does not allow
441                // null pointer
442                unsafe { libc::malloc(bytes_cap) }
443            } else {
444                unsafe { libc::realloc(self.ptr.cast(), bytes_cap) }
445            };
446            self.ptr = new_ptr.cast();
447            self.cap = new_cap;
448        }
449    }
450
451    pub fn push(&mut self, elem: T) {
452        if self.cap == self.len {
453            self.reserve(if self.cap == 0 { 4 } else { self.cap });
454        }
455        debug_assert!(self.cap > self.len);
456        unsafe {
457            self.ptr.offset(self.len.unwrapped_as()).write(elem);
458        }
459        self.len += 1;
460    }
461
462    pub fn extend<I>(&mut self, iter: I)
463    where
464        I: IntoIterator<Item = T>,
465    {
466        for item in iter {
467            self.push(item);
468        }
469    }
470}
471
472impl<T> Default for VecLike<T> {
473    fn default() -> Self {
474        VecLike::new()
475    }
476}
477
478impl<T> Drop for VecLike<T> {
479    fn drop(&mut self) {
480        // if capacity is zero, ptr is dangling
481        if self.cap == 0 {
482            return;
483        }
484        unsafe {
485            let s = ptr::slice_from_raw_parts_mut(self.ptr, self.len);
486            self.len = 0;
487            ptr::drop_in_place(s);
488            libc::free(self.ptr.cast());
489        }
490    }
491}
492
493impl<T> FromIterator<T> for VecLike<T> {
494    fn from_iter<I>(iter: I) -> Self
495    where
496        I: IntoIterator<Item = T>,
497    {
498        let mut vec = VecLike::new();
499        vec.extend(iter);
500        vec
501    }
502}
503
504impl<T> Extend<T> for VecLike<T> {
505    fn extend<I>(&mut self, iter: I)
506    where
507        I: IntoIterator<Item = T>,
508    {
509        self.extend(iter);
510    }
511}

513pub trait SameSizeAndAlign {
514    const CHECK_SIZE: ();
515    const CHECK_ALIGN: ();
516}
517
518impl<Src, Dst> SameSizeAndAlign for (Src, Dst) {
519    const CHECK_SIZE: () = assert!(mem::size_of::<Src>() == mem::size_of::<Dst>());
520    const CHECK_ALIGN: () = assert!(mem::align_of::<Src>() == mem::align_of::<Dst>());
521}
522
523#[allow(clippy::let_unit_value)]
524pub const fn cast_ptr<Src, Dst>(ptr: *const Src) -> *const Dst
525where
526    (Src, Dst): SameSizeAndAlign,
527{
528    // Force the size and alignment checks to be evaluated at compile time
529    let _check = <(Src, Dst) as SameSizeAndAlign>::CHECK_SIZE;
530    let _check = <(Src, Dst) as SameSizeAndAlign>::CHECK_ALIGN;
531
532    ptr.cast()
533}
534
535#[allow(clippy::let_unit_value)]
536pub const fn cast_ptr_mut<Src, Dst>(ptr: *mut Src) -> *mut Dst
537where
538    (Src, Dst): SameSizeAndAlign,
539{
540    // Force the size and alignment checks to be evaluated at compile time
541    let _check = <(Src, Dst) as SameSizeAndAlign>::CHECK_SIZE;
542    let _check = <(Src, Dst) as SameSizeAndAlign>::CHECK_ALIGN;
543
544    ptr.cast()
