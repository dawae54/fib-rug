36/**
37An arbitrary-precision rational number.
38
39A `Rational` number is made up of a numerator [`Integer`] and denominator
40[`Integer`]. After `Rational` number functions, the number is always in
41canonical form, that is the denominator is always greater than zero, and there
42are no common factors. Zero is stored as 0/1.
43
44# Examples
45
46```rust
47use rug::Rational;
48let r = Rational::from((-12, 15));
49let recip = Rational::from(r.recip_ref());
50assert_eq!(recip, Rational::from((-5, 4)));
51assert_eq!(recip, -1.25);
52// The numerator and denominator are stored in canonical form.
53let (num, den) = r.into_numer_denom();
54assert_eq!(num, -4);
55assert_eq!(den, 5);
56```
57
58When comparing `Rational` numbers, it is sometimes convenient to compare to a
59pair of primitive integers, like in the comparison to `Rational::from((-5, 4))`
60in the example above. `Rational::from((-5, 4))` requires memory allocation,
61which can be an expensive operation. To avoid this allocation, when both the
62numerator and denominator fit in a primitive integer, we can use
63[`MiniRational`] instead, which does not require heap memory allocation.
64
65```rust
66use rug::rational::MiniRational;
67use rug::Rational;
68let r = Rational::from((-12, 15));
69let recip = Rational::from(r.recip_ref());
70assert_eq!(recip, MiniRational::from((-5, 4)));
71```
72
73The `Rational` number type supports various functions. Most methods have three
74versions:
75
76 1. The first method consumes the operand.
77 2. The second method has a “`_mut`” suffix and mutates the operand.
78 3. The third method has a “`_ref`” suffix and borrows the operand. The returned
79    item is an [incomplete-computation value][icv] that can be assigned to a
80    `Rational` number.
81
82```rust
83use rug::rational::MiniRational;
84use rug::Rational;
85
86// 1. consume the operand
87let a = Rational::from((-15, 2));
88let abs_a = a.abs();
89assert_eq!(abs_a, MiniRational::from((15, 2)));
90
91// 2. mutate the operand
92let mut b = Rational::from((-17, 2));
93b.abs_mut();
94assert_eq!(b, MiniRational::from((17, 2)));
95
96// 3. borrow the operand
97let c = Rational::from((-19, 2));
98let r = c.abs_ref();
99let abs_c = Rational::from(r);
100assert_eq!(abs_c, MiniRational::from((19, 2)));
101// c was not consumed
102assert_eq!(c, MiniRational::from((-19, 2)));
103```
104
105[`MiniRational`]: crate::rational::MiniRational
106[icv]: crate#incomplete-computation-values
107*/

247    /// Creates a [`Rational`] number from an initialized [GMP rational
248    /// number][mpq_t].
249    ///
250    /// # Safety
251    ///
252    ///   * The function must *not* be used to create a constant [`Rational`]
253    ///     number, though it can be used to create a static [`Rational`]
254    ///     number. This is because constant values are *copied* on use, leading
255    ///     to undefined behavior when they are dropped.
256    ///   * The value must be initialized as a valid [`mpq_t`].
257    ///   * The [`mpq_t`] type can be considered as a kind of pointer, so there
258    ///     can be multiple copies of it. Since this function takes over
259    ///     ownership, no other copies of the passed value should exist.
260    ///   * The numerator and denominator must be in canonical form, as the rest
261    ///     of the library assumes that they are. Most GMP functions leave the
262    ///     rational number in canonical form, but assignment functions do not.
263    ///     Check the [GMP documentation][gmp mpq] for details.
264    ///
265    /// # Examples
266    ///
267    /// ```rust
268    /// use core::mem::MaybeUninit;
269    /// use gmp_mpfr_sys::gmp;
270    /// use rug::rational::MiniRational;
271    /// use rug::Rational;
272    /// let r = unsafe {
273    ///     let mut q = MaybeUninit::uninit();
274    ///     gmp::mpq_init(q.as_mut_ptr());
275    ///     let mut q = q.assume_init();
276    ///     gmp::mpq_set_si(&mut q, -145, 10);
277    ///     gmp::mpq_canonicalize(&mut q);
278    ///     // q is initialized and unique
279    ///     Rational::from_raw(q)
280    /// };
281    /// assert_eq!(r, MiniRational::from((-145, 10)));
282    /// // since r is a Rational now, deallocation is automatic
283    /// ```
284    ///
285    /// This can be used to create a static [`Rational`] number using
286    /// [`MPZ_ROINIT_N`][gmp::MPZ_ROINIT_N] to initialize the raw numerator and
287    /// denominator values. See the [GMP documentation][gmp roinit] for details.
288    ///
289    /// ```rust
290    /// use gmp_mpfr_sys::gmp;
291    /// use gmp_mpfr_sys::gmp::{limb_t, mpq_t};
292    /// use rug::{Integer, Rational};
293    /// const NUMER_LIMBS: [limb_t; 2] = [0, 5];
294    /// const DENOM_LIMBS: [limb_t; 1] = [3];
295    /// const MPQ: mpq_t = unsafe {
296    ///     mpq_t {
297    ///         num: gmp::MPZ_ROINIT_N(NUMER_LIMBS.as_ptr().cast_mut(), -2),
298    ///         den: gmp::MPZ_ROINIT_N(DENOM_LIMBS.as_ptr().cast_mut(), 1),
299    ///     }
300    /// };
301    /// // Must *not* be const, otherwise it would lead to undefined
302    /// // behavior on use, as it would create a copy that is dropped.
303    /// static R: Rational = unsafe { Rational::from_raw(MPQ) };
304    /// let numer_check =
305    ///     -((Integer::from(NUMER_LIMBS[1]) << gmp::NUMB_BITS) + NUMER_LIMBS[0]);
306    /// let denom_check = Integer::from(DENOM_LIMBS[0]);
307    /// assert_eq!(*R.numer(), numer_check);
308    /// assert_eq!(*R.denom(), denom_check);
309    /// let check = Rational::from((&numer_check, &denom_check));
310    /// assert_eq!(R, check);
311    /// assert_eq!(*R.numer(), *check.numer());
312    /// assert_eq!(*R.denom(), *check.denom());
313    /// ```
314    ///
315    /// [gmp mpq]: gmp_mpfr_sys::C::GMP::Rational_Number_Functions#index-Rational-number-functions
316    /// [gmp roinit]: gmp_mpfr_sys::C::GMP::Integer_Functions#index-MPZ_005fROINIT_005fN
317    #[inline]
318    pub const unsafe fn from_raw(raw: mpq_t) -> Self {
319        Rational { inner: raw }
320    }

857    /// Calls a function with mutable references to the numerator and
858    /// denominator, then canonicalizes the number.
859    ///
860    /// The denominator must not be zero when the function returns.
861    ///
862    /// # Panics
863    ///
864    /// Panics if the denominator is zero when the function returns.
865    ///
866    /// # Examples
867    ///
868    /// ```rust
869    /// use rug::Rational;
870    /// let mut r = Rational::from((3, 5));
871    /// r.mutate_numer_denom(|num, den| {
872    ///     // change r from 3/5 to 4/8, which is equal to 1/2
873    ///     *num += 1;
874    ///     *den += 3;
875    /// });
876    /// assert_eq!(*r.numer(), 1);
877    /// assert_eq!(*r.denom(), 2);
878    /// ```
879    ///
880    /// This method does not check that the numerator and denominator are in
881    /// canonical form before calling `func`. This means that this method can be
882    /// used to canonicalize the number after some unsafe methods that do not
883    /// leave the number in cononical form.
884    ///
885    /// ```rust
886    /// use rug::Rational;
887    /// let mut r = Rational::from((3, 5));
888    /// unsafe {
889    ///     // leave r in non-canonical form
890    ///     *r.as_mut_numer_denom_no_canonicalization().0 += 1;
891    ///     *r.as_mut_numer_denom_no_canonicalization().1 -= 13;
892    /// }
893    /// // At this point, r is still not canonical: 4 / -8
894    /// assert_eq!(*r.numer(), 4);
895    /// assert_eq!(*r.denom(), -8);
896    /// r.mutate_numer_denom(|_, _| {});
897    /// // Now r is in canonical form: -1 / 2
898    /// assert_eq!(*r.numer(), -1);
899    /// assert_eq!(*r.denom(), 2);
900    /// ```
901    pub fn mutate_numer_denom<F>(&mut self, func: F)
902    where
903        F: FnOnce(&mut Integer, &mut Integer),
904    {
905        struct CanonicalizeOnDrop<'a>(&'a mut Rational);
906
907        impl Drop for CanonicalizeOnDrop<'_> {
908            fn drop(&mut self) {
909                xmpq::canonicalize(self.0);
910            }
911        }
912
913        let guard = CanonicalizeOnDrop(self);
914        unsafe {
915            let (num, den) = xmpq::numref_denref(guard.0);
916            func(num, den);
917        }
918    }

921    /// aftwerwards.
922    ///
923    /// # Safety
924    ///
925    /// This function is unsafe because it does not canonicalize the
926    /// [`Rational`] number when the borrow ends. The caller must ensure that
927    /// the numerator and denominator are left in canonical form, as the rest of
928    /// the library assumes that they are.
929    ///
930    /// There are a few methods that can be called on [`Rational`] numbers that
931    /// are not in canonical form:
932    ///
933    ///   * [`numer`][Rational::numer] and [`denom`][Rational::denom], which
934    ///     treat the numerator and denominator separately
935    ///   * assignment methods, which overwrite the previous value and leave the
936    ///     number in canonical form
937    ///   * [`mutate_numer_denom`][Rational::mutate_numer_denom], which treats
938    ///     the numerator and denominator seprarately, and leaves the number in
939    ///     canoncial form

968    ///     // second operation: subtract 13 from denominator
969    ///     *r.as_mut_numer_denom_no_canonicalization().1 -= 13;

973    /// assert_eq!(*r.denom(), -8);
974    /// r.mutate_numer_denom(|_, _| {});

976    /// assert_eq!(*r.numer(), -1);
977    /// assert_eq!(*r.denom(), 2);
978    /// ```
979    #[inline]
980    pub unsafe fn as_mut_numer_denom_no_canonicalization(
981        &mut self,
982    ) -> (&mut Integer, &mut Integer) {
983        unsafe { xmpq::numref_denref(self) }
984    }

3082    let bradix = radix.unwrapped_as::<u8>();

3086    let mut has_sign = false;
3087    let mut is_negative = false;
3088    let mut has_digits = false;

3092            if den_start.is_some() {

3096            }

3098                return Err(Error {
3099                    kind: Kind::NumerNoDigits,
3100                });

3102            has_digits = false;
3103            den_start = Some(digits.as_slice().len());
3104            continue;
3105        }
3106        let digit = match b {
3107            b'+' if den_start.is_none() && !has_sign && !has_digits => {
3108                has_sign = true;
3109                continue;
3110            }
3111            b'-' if den_start.is_none() && !has_sign && !has_digits => {
3112                is_negative = true;
3113                has_sign = true;
3114                continue;
3115            }
3116            b'_' if has_digits => continue,
3117            b' ' | b'\t' | b'\n' | 0x0b | 0x0c | 0x0d => continue,

3119            b'0'..=b'9' => b - b'0',

3121            b'A'..=b'Z' => b - b'A' + 10,
3122
3123            // error
3124            _ => bradix,
3125        };
3126        if digit >= bradix {
3127            return Err(Error {
3128                kind: Kind::InvalidDigit,
3129            });
3130        }
3131        has_digits = true;
3132        if digit > 0
3133            || (!digits.as_slice().is_empty() && den_start != Some(digits.as_slice().len()))
3134        {

3136        }
3137    }
3138    if !has_digits {

3152    let den_start = den_start.unwrap_or(digits.as_slice().len());
